48 CGLM_INLINE float glm_bezier(float s, float p0, float c0, float c1, float p1) {
49  float x, xx, ss, xs3, a;
50 
51  x = 1.0f - s;
52  xx = x * x;
53  ss = s * s;
54  xs3 = (s - ss) * 3.0f;
55  a = p0 * xx + c0 * xs3;
56 
57  return a + s * (c1 * xs3 + p1 * ss - a);
58 }

80 CGLM_INLINE float glm_hermite(float s, float p0, float t0, float t1, float p1) {
81  float ss, d, a, b, c, e, f;
82 
83  ss = s * s;
84  a = ss + ss;
85  c = a + ss;
86  b = a * s;
87  d = s * ss;
88  f = d - ss;
89  e = b - c;
90 
91  return p0 * (e + 1.0f) + t0 * (f - ss + s) + t1 * f - p1 * e;
92 }

106  float glm_decasteljau(float prm, float p0, float c0, float c1, float p1) {
107  float u, v, a, b, c, d, e, f;
108  int i;
109 
110  if (prm - p0 < CGLM_DECASTEL_SMALL)
111  return 0.0f;
112 
113  if (p1 - prm < CGLM_DECASTEL_SMALL)
114  return 1.0f;
115 
116  u = 0.0f;
117  v = 1.0f;
118 
119  for (i = 0; i < CGLM_DECASTEL_MAX; i++) {
120  /* de Casteljau Subdivision */
121  a = (p0 + c0) * 0.5f;
122  b = (c0 + c1) * 0.5f;
123  c = (c1 + p1) * 0.5f;
124  d = (a + b) * 0.5f;
125  e = (b + c) * 0.5f;
126  f = (d + e) * 0.5f; /* this one is on the curve! */
127 
128  /* The curve point is close enough to our wanted t */
129  if (fabsf(f - prm) < CGLM_DECASTEL_EPS)
130  return glm_clamp_zo((u + v) * 0.5f);
131 
132  /* dichotomy */
133  if (f < prm) {
134  p0 = f;
135  c0 = e;
136  c1 = c;
137  u = (u + v) * 0.5f;
138  } else {
139  c0 = a;
140  c1 = d;
141  p1 = f;
142  v = (u + v) * 0.5f;
143  }
144  }
145 
146  return glm_clamp_zo((u + v) * 0.5f);
147 }
