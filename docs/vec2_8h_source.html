426 CGLM_INLINE void glm_vec2_rotate(vec2 v, float angle, vec2 dest) {
427  float c, s, x1, y1;
428 
429  c = cosf(angle);
430  s = sinf(angle);
431 
432  x1 = v[0];
433  y1 = v[1];
434 
435  dest[0] = c * x1 - s * y1;
436  dest[1] = s * x1 + c * y1;

507 CGLM_INLINE void glm_vec2_lerp(vec2 from, vec2 to, float t, vec2 dest) {
508  vec2 s, v;
509 
510  /* from + s * (to - from) */
511  glm_vec2_fill(s, glm_clamp_zo(t));
512  glm_vec2_sub(to, from, v);
513  glm_vec2_mul(s, v, v);
514  glm_vec2_add(from, v, dest);
515 }
