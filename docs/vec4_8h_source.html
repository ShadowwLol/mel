633 CGLM_INLINE void glm_vec4_normalize_to(vec4 v, vec4 dest) {
634 #if defined( __SSE__ ) || defined( __SSE2__ )
635  __m128 xdot, x0;
636  float dot;
637 
638  x0 = glmm_load(v);
639  xdot = glmm_vdot(x0, x0);
640  dot = _mm_cvtss_f32(xdot);
641 
642  if (dot == 0.0f) {
643  glmm_store(dest, _mm_setzero_ps());
644  return;
645  }
646 
647  glmm_store(dest, _mm_div_ps(x0, _mm_sqrt_ps(xdot)));
648 #else
649  float norm;
650 
651  norm = glm_vec4_norm(v);
652 
653  if (norm == 0.0f) {
654  glm_vec4_zero(dest);
655  return;
656  }
657 
658  glm_vec4_scale(v, 1.0f / norm, dest);
659 #endif
660 }
