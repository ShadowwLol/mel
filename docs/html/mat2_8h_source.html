124 CGLM_INLINE void glm_mat2_mul(mat2 m1, mat2 m2, mat2 dest) {
125 #if defined( __SSE__ ) || defined( __SSE2__ )
126  glm_mat2_mul_sse2(m1, m2, dest);
127 #elif defined(CGLM_NEON_FP)
128  glm_mat2_mul_neon(m1, m2, dest);
129 #else
130  float a00 = m1[0][0], a01 = m1[0][1],
131  a10 = m1[1][0], a11 = m1[1][1],
132  b00 = m2[0][0], b01 = m2[0][1], b10 = m2[1][0], b11 = m2[1][1];
133 
134  dest[0][0] = a00 * b00 + a10 * b01;
135  dest[0][1] = a01 * b00 + a11 * b01;
136  dest[1][0] = a00 * b10 + a10 * b11;
137  dest[1][1] = a01 * b10 + a11 * b11;
138 #endif
139 }

203 CGLM_INLINE void glm_mat2_scale(mat2 m, float s) {
204 #if defined( __SSE__ ) || defined( __SSE2__ )
205  glmm_store(m[0], _mm_mul_ps(_mm_loadu_ps(m[0]), _mm_set1_ps(s)));
206 #elif defined(CGLM_NEON_FP)
207  vst1q_f32(m[0], vmulq_f32(vld1q_f32(m[0]), vdupq_n_f32(s)));
208 #else
209  m[0][0] = m[0][0] * s;
210  m[0][1] = m[0][1] * s;
211  m[1][0] = m[1][0] * s;
212  m[1][1] = m[1][1] * s;
213 #endif
214 }

233 CGLM_INLINE void glm_mat2_inv(mat2 mat, mat2 dest) {
234  float det;
235  float a = mat[0][0], b = mat[0][1], c = mat[1][0], d = mat[1][1];
236 
237  det = 1.0f / (a * d - b * c);
238 
239  dest[0][0] = d * det;
240  dest[0][1] = -b * det;
241  dest[1][0] = -c * det;
242  dest[1][1] = a * det;
243 }
