64 CGLM_INLINE
65  void
66 glm_frustum_rh_zo(float left, float right,
67  float bottom, float top,
68  float nearZ, float farZ, mat4 dest) {
69  float rl, tb, fn, nv;
70 
71  glm_mat4_zero(dest);
72 
73  rl = 1.0f / (right - left);
74  tb = 1.0f / (top - bottom);
75  fn = -1.0f / (farZ - nearZ);
76  nv = 2.0f * nearZ;
77 
78  dest[0][0] = nv * rl;
79  dest[1][1] = nv * tb;
80  dest[2][0] = (right + left) * rl;
81  dest[2][1] = (top + bottom) * tb;
82  dest[2][2] = farZ * fn;
83  dest[2][3] = -1.0f;
84  dest[3][2] = farZ * nearZ * fn;
85 }

98  void
99 glm_perspective_rh_zo(float fovy,
100  float aspect, float nearZ, float farZ, mat4 dest) {
101  float f, fn;
102 
103  glm_mat4_zero(dest);
104 
105  f = 1.0f / tanf(fovy * 0.5f);
106  fn = 1.0f / (nearZ - farZ);
107 
108  dest[0][0] = f / aspect;
109  dest[1][1] = f;
110  dest[2][2] = farZ * fn;
111  dest[2][3] = -1.0f;
112  dest[3][2] = nearZ * farZ * fn;
113 }

152 CGLM_INLINE void glm_persp_move_far_rh_zo(mat4 proj, float deltaFar) {
153  float fn, farZ, nearZ, p22, p32;
154 
155  p22 = proj[2][2];
156  p32 = proj[3][2];
157 
158  nearZ = p32 / p22;
159  farZ = p32 / (p22 + 1.0f) + deltaFar;
160  fn = 1.0f / (nearZ - farZ);
161 
162  proj[2][2] = farZ * fn;
163  proj[3][2] = nearZ * farZ * fn;
164 }

179 CGLM_INLINE
180  void
181 glm_persp_decomp_rh_zo(mat4 proj,
182  float* __restrict nearZ, float* __restrict farZ,
183  float* __restrict top, float* __restrict bottom,
184  float* __restrict left, float* __restrict right) {
185  float m00, m11, m20, m21, m22, m32, n, f;
186  float n_m11, n_m00;
187 
188  m00 = proj[0][0];
189  m11 = proj[1][1];
190  m20 = proj[2][0];
191  m21 = proj[2][1];
192  m22 = proj[2][2];
193  m32 = proj[3][2];
194 
195  n = m32 / m22;
196  f = m32 / (m22 + 1.0f);
197 
198  n_m11 = n / m11;
199  n_m00 = n / m00;
200 
201  *nearZ = n;
202  *farZ = f;
203  *bottom = n_m11 * (m21 - 1.0f);
204  *top = n_m11 * (m21 + 1.0f);
205  *left = n_m00 * (m20 - 1.0f);
206  *right = n_m00 * (m20 + 1.0f);
207 }

233 glm_persp_decomp_x_rh_zo(mat4 proj,
234  float* __restrict left, float* __restrict right) {
235  float nearZ, m20, m00, m22;
236 
237  m00 = proj[0][0];
238  m20 = proj[2][0];
239  m22 = proj[2][2];
240

242  *left = nearZ * (m20 - 1.0f) / m00;
243  *right = nearZ * (m20 + 1.0f) / m00;

328 CGLM_INLINE void glm_persp_sizes_rh_zo(mat4 proj, float fovy, vec4 dest) {
329  float t, a, nearZ, farZ;
330 
331  t = 2.0f * tanf(fovy * 0.5f);
332  a = glm_persp_aspect(proj);
333 
334  glm_persp_decomp_z_rh_zo(proj, &nearZ, &farZ);
335 
336  dest[1] = t * nearZ;
337  dest[3] = t * farZ;
338  dest[0] = a * dest[1];
339  dest[2] = a * dest[3];
340 }
