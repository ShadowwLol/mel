290 #ifdef __AVX__
291  glm_mat4_mul_avx(m1, m2, dest);
292 #elif defined( __SSE__ ) || defined( __SSE2__ )
293  glm_mat4_mul_sse2(m1, m2, dest);
294 #elif defined(CGLM_NEON_FP)
295  glm_mat4_mul_neon(m1, m2, dest);
296 #else
297  float a00 = m1[0][0], a01 = m1[0][1], a02 = m1[0][2], a03 = m1[0][3],
298  a10 = m1[1][0], a11 = m1[1][1], a12 = m1[1][2], a13 = m1[1][3],
299  a20 = m1[2][0], a21 = m1[2][1], a22 = m1[2][2], a23 = m1[2][3],
300  a30 = m1[3][0], a31 = m1[3][1], a32 = m1[3][2], a33 = m1[3][3],
301  b00 = m2[0][0], b01 = m2[0][1], b02 = m2[0][2], b03 = m2[0][3],
302  b10 = m2[1][0], b11 = m2[1][1], b12 = m2[1][2], b13 = m2[1][3],
303  b20 = m2[2][0], b21 = m2[2][1], b22 = m2[2][2], b23 = m2[2][3],
304  b30 = m2[3][0], b31 = m2[3][1], b32 = m2[3][2], b33 = m2[3][3];
305 
306  dest[0][0] = a00 * b00 + a10 * b01 + a20 * b02 + a30 * b03;
307  dest[0][1] = a01 * b00 + a11 * b01 + a21 * b02 + a31 * b03;
308  dest[0][2] = a02 * b00 + a12 * b01 + a22 * b02 + a32 * b03;
309  dest[0][3] = a03 * b00 + a13 * b01 + a23 * b02 + a33 * b03;
310  dest[1][0] = a00 * b10 + a10 * b11 + a20 * b12 + a30 * b13;
311  dest[1][1] = a01 * b10 + a11 * b11 + a21 * b12 + a31 * b13;
312  dest[1][2] = a02 * b10 + a12 * b11 + a22 * b12 + a32 * b13;
313  dest[1][3] = a03 * b10 + a13 * b11 + a23 * b12 + a33 * b13;
314  dest[2][0] = a00 * b20 + a10 * b21 + a20 * b22 + a30 * b23;
315  dest[2][1] = a01 * b20 + a11 * b21 + a21 * b22 + a31 * b23;
316  dest[2][2] = a02 * b20 + a12 * b21 + a22 * b22 + a32 * b23;
317  dest[2][3] = a03 * b20 + a13 * b21 + a23 * b22 + a33 * b23;
318  dest[3][0] = a00 * b30 + a10 * b31 + a20 * b32 + a30 * b33;
319  dest[3][1] = a01 * b30 + a11 * b31 + a21 * b32 + a31 * b33;
320  dest[3][2] = a02 * b30 + a12 * b31 + a22 * b32 + a32 * b33;
321  dest[3][3] = a03 * b30 + a13 * b31 + a23 * b32 + a33 * b33;
322 #endif
323 }

408 CGLM_INLINE void glm_mat4_quat(mat4 m, versor dest) {
409  float trace, r, rinv;
410 
411  /* it seems using like m12 instead of m[1][2] causes extra instructions */
412 
413  trace = m[0][0] + m[1][1] + m[2][2];
414  if (trace >= 0.0f) {
415  r = sqrtf(1.0f + trace);
416  rinv = 0.5f / r;
417 
418  dest[0] = rinv * (m[1][2] - m[2][1]);
419  dest[1] = rinv * (m[2][0] - m[0][2]);
420  dest[2] = rinv * (m[0][1] - m[1][0]);
421  dest[3] = r * 0.5f;
422  } else if (m[0][0] >= m[1][1] && m[0][0] >= m[2][2]) {
423  r = sqrtf(1.0f - m[1][1] - m[2][2] + m[0][0]);
424  rinv = 0.5f / r;
425 
426  dest[0] = r * 0.5f;
427  dest[1] = rinv * (m[0][1] + m[1][0]);
428  dest[2] = rinv * (m[0][2] + m[2][0]);
429  dest[3] = rinv * (m[1][2] - m[2][1]);
430  } else if (m[1][1] >= m[2][2]) {
431  r = sqrtf(1.0f - m[0][0] - m[2][2] + m[1][1]);
432  rinv = 0.5f / r;
433 
434  dest[0] = rinv * (m[0][1] + m[1][0]);
435  dest[1] = r * 0.5f;
436  dest[2] = rinv * (m[1][2] + m[2][1]);
437  dest[3] = rinv * (m[2][0] - m[0][2]);
438  } else {
439  r = sqrtf(1.0f - m[0][0] - m[1][1] + m[2][2]);
440  rinv = 0.5f / r;
441 
442  dest[0] = rinv * (m[0][2] + m[2][0]);
443  dest[1] = rinv * (m[1][2] + m[2][1]);
444  dest[2] = r * 0.5f;
445  dest[3] = rinv * (m[0][1] - m[1][0]);
446  }
447 }

571 CGLM_INLINE float glm_mat4_det(mat4 mat) {
572 #if defined( __SSE__ ) || defined( __SSE2__ )
573  return glm_mat4_det_sse2(mat);
574 #elif defined(CGLM_NEON_FP)
575  return glm_mat4_det_neon(mat);
576 #else
577  /* [square] det(A) = det(At) */
578  float t[6];
579  float a = mat[0][0], b = mat[0][1], c = mat[0][2], d = mat[0][3],
580  e = mat[1][0], f = mat[1][1], g = mat[1][2], h = mat[1][3],
581  i = mat[2][0], j = mat[2][1], k = mat[2][2], l = mat[2][3],
582  m = mat[3][0], n = mat[3][1], o = mat[3][2], p = mat[3][3];
583 
584  t[0] = k * p - o * l;
585  t[1] = j * p - n * l;
586  t[2] = j * o - n * k;
587  t[3] = i * p - m * l;
588  t[4] = i * o - m * k;
589  t[5] = i * n - m * j;
590 
591  return a * (f * t[0] - g * t[1] + h * t[2])
592  - b * (e * t[0] - g * t[3] + h * t[4])
593  + c * (e * t[1] - f * t[3] + h * t[5])
594  - d * (e * t[2] - f * t[4] + g * t[5]);
595 #endif
596 }

604 CGLM_INLINE void glm_mat4_inv(mat4 mat, mat4 dest) {
605 #if defined( __SSE__ ) || defined( __SSE2__ )
606  glm_mat4_inv_sse2(mat, dest);
607 #elif defined(CGLM_NEON_FP)
608  glm_mat4_inv_neon(mat, dest);
609 #else
610  float t[6];
611  float det;
612  float a = mat[0][0], b = mat[0][1], c = mat[0][2], d = mat[0][3],
613  e = mat[1][0], f = mat[1][1], g = mat[1][2], h = mat[1][3],
614  i = mat[2][0], j = mat[2][1], k = mat[2][2], l = mat[2][3],
615  m = mat[3][0], n = mat[3][1], o = mat[3][2], p = mat[3][3];
616 
617  t[0] = k * p - o * l;
618  t[1] = j * p - n * l;
619  t[2] = j * o - n * k;
620  t[3] = i * p - m * l;
621  t[4] = i * o - m * k;
622  t[5] = i * n - m * j;
623 
624  dest[0][0] = f * t[0] - g * t[1] + h * t[2];
625  dest[1][0] = -(e * t[0] - g * t[3] + h * t[4]);
626  dest[2][0] = e * t[1] - f * t[3] + h * t[5];
627  dest[3][0] = -(e * t[2] - f * t[4] + g * t[5]);
628 
629  dest[0][1] = -(b * t[0] - c * t[1] + d * t[2]);
630  dest[1][1] = a * t[0] - c * t[3] + d * t[4];
631  dest[2][1] = -(a * t[1] - b * t[3] + d * t[5]);
632  dest[3][1] = a * t[2] - b * t[4] + c * t[5];
633 
634  t[0] = g * p - o * h;
635  t[1] = f * p - n * h;
636  t[2] = f * o - n * g;
637  t[3] = e * p - m * h;
