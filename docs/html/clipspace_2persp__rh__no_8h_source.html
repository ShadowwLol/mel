65 CGLM_INLINE
66  void
67 glm_frustum_rh_no(float left, float right,
68  float bottom, float top,
69  float nearZ, float farZ, mat4 dest) {
70  float rl, tb, fn, nv;
71 
72  glm_mat4_zero(dest);
73 
74  rl = 1.0f / (right - left);
75  tb = 1.0f / (top - bottom);
76  fn = -1.0f / (farZ - nearZ);
77  nv = 2.0f * nearZ;
78 
79  dest[0][0] = nv * rl;
80  dest[1][1] = nv * tb;
81  dest[2][0] = (right + left) * rl;
82  dest[2][1] = (top + bottom) * tb;
83  dest[2][2] = (farZ + nearZ) * fn;
84  dest[2][3] = -1.0f;
85  dest[3][2] = farZ * nv * fn;
86 }

100  void
101 glm_perspective_rh_no(float fovy,
102  float aspect, float nearZ, float farZ, mat4 dest) {
103  float f, fn;
104 
105  glm_mat4_zero(dest);
106 
107  f = 1.0f / tanf(fovy * 0.5f);
108  fn = 1.0f / (nearZ - farZ);
109 
110  dest[0][0] = f / aspect;
111  dest[1][1] = f;
112  dest[2][2] = (nearZ + farZ) * fn;
113  dest[2][3] = -1.0f;
114  dest[3][2] = 2.0f * nearZ * farZ * fn;
115 
116 }

156 CGLM_INLINE void glm_persp_move_far_rh_no(mat4 proj, float deltaFar) {
157  float fn, farZ, nearZ, p22, p32;
158 
159  p22 = proj[2][2];
160  p32 = proj[3][2];
161 
162  nearZ = p32 / (p22 - 1.0f);
163  farZ = p32 / (p22 + 1.0f) + deltaFar;
164  fn = 1.0f / (nearZ - farZ);
165 
166  proj[2][2] = (farZ + nearZ) * fn;
167  proj[3][2] = 2.0f * nearZ * farZ * fn;
168 }

184  void
185 glm_persp_decomp_rh_no(mat4 proj,
186  float* __restrict nearZ, float* __restrict farZ,
187  float* __restrict top, float* __restrict bottom,
188  float* __restrict left, float* __restrict right) {
189  float m00, m11, m20, m21, m22, m32, n, f;
190  float n_m11, n_m00;
191 
192  m00 = proj[0][0];
193  m11 = proj[1][1];
194  m20 = proj[2][0];
195  m21 = proj[2][1];
196  m22 = proj[2][2];
197  m32 = proj[3][2];
198 
199  n = m32 / (m22 - 1.0f);
200  f = m32 / (m22 + 1.0f);
201 
202  n_m11 = n / m11;
203  n_m00 = n / m00;
204 
205  *nearZ = n;
206  *farZ = f;
207  *bottom = n_m11 * (m21 - 1.0f);
208  *top = n_m11 * (m21 + 1.0f);
209  *left = n_m00 * (m20 - 1.0f);
210  *right = n_m00 * (m20 + 1.0f);
211 }

334 CGLM_INLINE void glm_persp_sizes_rh_no(mat4 proj, float fovy, vec4 dest) {
335  float t, a, nearZ, farZ;
336 
337  t = 2.0f * tanf(fovy * 0.5f);
338  a = glm_persp_aspect(proj);
339 
340  glm_persp_decomp_z_rh_no(proj, &nearZ, &farZ);
341 
342  dest[1] = t * nearZ;
343  dest[3] = t * farZ;
344  dest[0] = a * dest[1];
345  dest[2] = a * dest[3];
346 }
