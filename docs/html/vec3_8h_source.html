522 CGLM_INLINE void glm_vec3_cross(vec3 a, vec3 b, vec3 dest) {
523  vec3 c;
524  /* (u2.v3 - u3.v2, u3.v1 - u1.v3, u1.v2 - u2.v1) */
525  c[0] = a[1] * b[2] - a[2] * b[1];
526  c[1] = a[2] * b[0] - a[0] * b[2];
527  c[2] = a[0] * b[1] - a[1] * b[0];
528  glm_vec3_copy(c, dest);
529 }

551 CGLM_INLINE float glm_vec3_angle(vec3 a, vec3 b) {
552  float norm, dot;
553 
554  /* maybe compiler generate approximation instruction (rcp) */
555  norm = 1.0f / (glm_vec3_norm(a) * glm_vec3_norm(b));
556  dot = glm_vec3_dot(a, b) * norm;
557 
558  if (dot > 1.0f)
559  return 0.0f;
560  else if (dot < -1.0f)
561  return CGLM_PI;
562 
563  return acosf(dot);
564 }

573 CGLM_INLINE void glm_vec3_rotate(vec3 v, float angle, vec3 axis) {
574  vec3 v1, v2, k;
575  float c, s;
576 
577  c = cosf(angle);
578  s = sinf(angle);
579 
580  glm_vec3_normalize_to(axis, k);
581 
582  /* Right Hand, Rodrigues' rotation formula:
583  v = v*cos(t) + (kxv)sin(t) + k*(k.v)(1 - cos(t))
584  */
585  glm_vec3_scale(v, c, v1);
586 
587  glm_vec3_cross(k, v, v2);
588  glm_vec3_scale(v2, s, v2);
589 
590  glm_vec3_add(v1, v2, v1);
591 
592  glm_vec3_scale(k, glm_vec3_dot(k, v) * (1.0f - c), v2);
593  glm_vec3_add(v1, v2, v);
594 }
