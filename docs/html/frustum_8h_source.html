70 CGLM_INLINE void glm_frustum_planes(mat4 m, vec4 dest[6]) {
71  mat4 t;
72 
73  glm_mat4_transpose_to(m, t);
74 
75  glm_vec4_add(t[3], t[0], dest[0]); /* left */
76  glm_vec4_sub(t[3], t[0], dest[1]); /* right */
77  glm_vec4_add(t[3], t[1], dest[2]); /* bottom */
78  glm_vec4_sub(t[3], t[1], dest[3]); /* top */
79  glm_vec4_add(t[3], t[2], dest[4]); /* near */
80  glm_vec4_sub(t[3], t[2], dest[5]); /* far */
81 
82  glm_plane_normalize(dest[0]);
83  glm_plane_normalize(dest[1]);
84  glm_plane_normalize(dest[2]);
85  glm_plane_normalize(dest[3]);
86  glm_plane_normalize(dest[4]);
87  glm_plane_normalize(dest[5]);
88 }

113 CGLM_INLINE void glm_frustum_corners(mat4 invMat, vec4 dest[8]) {
114  vec4 c[8];
115 
116  /* indexOf(nearCoord) = indexOf(farCoord) + 4 */
117  vec4 csCoords[8] = {
118  GLM_CSCOORD_LBN,
119  GLM_CSCOORD_LTN,
120  GLM_CSCOORD_RTN,
121  GLM_CSCOORD_RBN,
122 
123  GLM_CSCOORD_LBF,
124  GLM_CSCOORD_LTF,
125  GLM_CSCOORD_RTF,
126  GLM_CSCOORD_RBF
127  };
128 
129  glm_mat4_mulv(invMat, csCoords[0], c[0]);
130  glm_mat4_mulv(invMat, csCoords[1], c[1]);
131  glm_mat4_mulv(invMat, csCoords[2], c[2]);
132  glm_mat4_mulv(invMat, csCoords[3], c[3]);
133  glm_mat4_mulv(invMat, csCoords[4], c[4]);
134  glm_mat4_mulv(invMat, csCoords[5], c[5]);
135  glm_mat4_mulv(invMat, csCoords[6], c[6]);
136  glm_mat4_mulv(invMat, csCoords[7], c[7]);
137 
138  glm_vec4_scale(c[0], 1.0f / c[0][3], dest[0]);
139  glm_vec4_scale(c[1], 1.0f / c[1][3], dest[1]);
140  glm_vec4_scale(c[2], 1.0f / c[2][3], dest[2]);
141  glm_vec4_scale(c[3], 1.0f / c[3][3], dest[3]);
142  glm_vec4_scale(c[4], 1.0f / c[4][3], dest[4]);
143  glm_vec4_scale(c[5], 1.0f / c[5][3], dest[5]);
144  glm_vec4_scale(c[6], 1.0f / c[6][3], dest[6]);
145  glm_vec4_scale(c[7], 1.0f / c[7][3], dest[7]);
146 }

177 CGLM_INLINE void glm_frustum_box(vec4 corners[8], mat4 m, vec3 box[2]) {
178  vec4 v;
179  vec3 min, max;
180  int i;
181 
182  glm_vec3_broadcast(FLT_MAX, min);
183  glm_vec3_broadcast(-FLT_MAX, max);
184 
185  for (i = 0; i < 8; i++) {
186  glm_mat4_mulv(m, corners[i], v);
187 
188  min[0] = glm_min(min[0], v[0]);
189  min[1] = glm_min(min[1], v[1]);
190  min[2] = glm_min(min[2], v[2]);
191 
192  max[0] = glm_max(max[0], v[0]);
193  max[1] = glm_max(max[1], v[1]);
194  max[2] = glm_max(max[2], v[2]);
195  }
196 
197  glm_vec3_copy(min, box[0]);
198  glm_vec3_copy(max, box[1]);
199 }

213 CGLM_INLINE
214  void
215 glm_frustum_corners_at(vec4 corners[8],
216  float splitDist, float farDist, vec4 planeCorners[4]) {
217  vec4 corner;
218  float dist, sc;
219 
220  /* because distance and scale is same for all */
221  dist = glm_vec3_distance(corners[GLM_RTF], corners[GLM_RTN]);
222  sc = dist * (splitDist / farDist);
223 
224  /* left bottom */
225  glm_vec4_sub(corners[GLM_LBF], corners[GLM_LBN], corner);
226  glm_vec4_scale_as(corner, sc, corner);
227  glm_vec4_add(corners[GLM_LBN], corner, planeCorners[0]);
228 
229  /* left top */
230  glm_vec4_sub(corners[GLM_LTF], corners[GLM_LTN], corner);
231  glm_vec4_scale_as(corner, sc, corner);
232  glm_vec4_add(corners[GLM_LTN], corner, planeCorners[1]);
233 
234  /* right top */
235  glm_vec4_sub(corners[GLM_RTF], corners[GLM_RTN], corner);
236  glm_vec4_scale_as(corner, sc, corner);
237  glm_vec4_add(corners[GLM_RTN], corner, planeCorners[2]);
238 
239  /* right bottom */
240  glm_vec4_sub(corners[GLM_RBF], corners[GLM_RBN], corner);
241  glm_vec4_scale_as(corner, sc, corner);
242  glm_vec4_add(corners[GLM_RBN], corner, planeCorners[3]);
243 }
