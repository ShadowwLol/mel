49 CGLM_INLINE void glm_mul(mat4 m1, mat4 m2, mat4 dest) {
50 #ifdef __AVX__
51  glm_mul_avx(m1, m2, dest);
52 #elif defined( __SSE__ ) || defined( __SSE2__ )
53  glm_mul_sse2(m1, m2, dest);
54 #elif defined(CGLM_NEON_FP)
55  glm_mul_neon(m1, m2, dest);
56 #else
57  float a00 = m1[0][0], a01 = m1[0][1], a02 = m1[0][2], a03 = m1[0][3],
58  a10 = m1[1][0], a11 = m1[1][1], a12 = m1[1][2], a13 = m1[1][3],
59  a20 = m1[2][0], a21 = m1[2][1], a22 = m1[2][2], a23 = m1[2][3],
60  a30 = m1[3][0], a31 = m1[3][1], a32 = m1[3][2], a33 = m1[3][3],
61  b00 = m2[0][0], b01 = m2[0][1], b02 = m2[0][2],
62  b10 = m2[1][0], b11 = m2[1][1], b12 = m2[1][2],
63  b20 = m2[2][0], b21 = m2[2][1], b22 = m2[2][2],
64  b30 = m2[3][0], b31 = m2[3][1], b32 = m2[3][2], b33 = m2[3][3];
65 
66  dest[0][0] = a00 * b00 + a10 * b01 + a20 * b02;
67  dest[0][1] = a01 * b00 + a11 * b01 + a21 * b02;
68  dest[0][2] = a02 * b00 + a12 * b01 + a22 * b02;
69  dest[0][3] = a03 * b00 + a13 * b01 + a23 * b02;
70 
71  dest[1][0] = a00 * b10 + a10 * b11 + a20 * b12;
72  dest[1][1] = a01 * b10 + a11 * b11 + a21 * b12;
73  dest[1][2] = a02 * b10 + a12 * b11 + a22 * b12;
74  dest[1][3] = a03 * b10 + a13 * b11 + a23 * b12;
75 
76  dest[2][0] = a00 * b20 + a10 * b21 + a20 * b22;
77  dest[2][1] = a01 * b20 + a11 * b21 + a21 * b22;
78  dest[2][2] = a02 * b20 + a12 * b21 + a22 * b22;
79  dest[2][3] = a03 * b20 + a13 * b21 + a23 * b22;
80 
81  dest[3][0] = a00 * b30 + a10 * b31 + a20 * b32 + a30 * b33;
82  dest[3][1] = a01 * b30 + a11 * b31 + a21 * b32 + a31 * b33;
83  dest[3][2] = a02 * b30 + a12 * b31 + a22 * b32 + a32 * b33;
84  dest[3][3] = a03 * b30 + a13 * b31 + a23 * b32 + a33 * b33;
85 #endif
86 }

150 CGLM_INLINE void glm_inv_tr(mat4 mat) {
151 #if defined( __SSE__ ) || defined( __SSE2__ )
152  glm_inv_tr_sse2(mat);
153 #elif defined(CGLM_NEON_FP)
154  glm_inv_tr_neon(mat);
155 #else
156  CGLM_ALIGN_MAT mat3 r;
157  CGLM_ALIGN(8) vec3 t;
158 
159  /* rotate */
160  glm_mat4_pick3t(mat, r);
161  glm_mat4_ins3(r, mat);
162 
163  /* translate */
164  glm_mat3_mulv(r, mat[3], t);
165  glm_vec3_negate(t);
166  glm_vec3_copy(t, mat[3]);
167 #endif
168 }
