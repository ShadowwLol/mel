140 CGLM_INLINE void glm_mat3_mul(mat3 m1, mat3 m2, mat3 dest) {
141 #if defined( __SSE__ ) || defined( __SSE2__ )
142  glm_mat3_mul_sse2(m1, m2, dest);
143 #else
144  float a00 = m1[0][0], a01 = m1[0][1], a02 = m1[0][2],
145  a10 = m1[1][0], a11 = m1[1][1], a12 = m1[1][2],
146  a20 = m1[2][0], a21 = m1[2][1], a22 = m1[2][2],
147  b00 = m2[0][0], b01 = m2[0][1], b02 = m2[0][2],
148  b10 = m2[1][0], b11 = m2[1][1], b12 = m2[1][2],
149  b20 = m2[2][0], b21 = m2[2][1], b22 = m2[2][2];
150 
151  dest[0][0] = a00 * b00 + a10 * b01 + a20 * b02;
152  dest[0][1] = a01 * b00 + a11 * b01 + a21 * b02;
153  dest[0][2] = a02 * b00 + a12 * b01 + a22 * b02;
154  dest[1][0] = a00 * b10 + a10 * b11 + a20 * b12;
155  dest[1][1] = a01 * b10 + a11 * b11 + a21 * b12;
156  dest[1][2] = a02 * b10 + a12 * b11 + a22 * b12;
157  dest[2][0] = a00 * b20 + a10 * b21 + a20 * b22;
158  dest[2][1] = a01 * b20 + a11 * b21 + a21 * b22;
159  dest[2][2] = a02 * b20 + a12 * b21 + a22 * b22;
160 #endif
161 }

213 CGLM_INLINE void glm_mat3_mulv(mat3 m, vec3 v, vec3 dest) {
214  vec3 res;

238 CGLM_INLINE void glm_mat3_quat(mat3 m, versor dest) {
239  float trace, r, rinv;
240 
241  /* it seems using like m12 instead of m[1][2] causes extra instructions */
242 
243  trace = m[0][0] + m[1][1] + m[2][2];
244  if (trace >= 0.0f) {
245  r = sqrtf(1.0f + trace);
246  rinv = 0.5f / r;
247 
248  dest[0] = rinv * (m[1][2] - m[2][1]);
249  dest[1] = rinv * (m[2][0] - m[0][2]);
250  dest[2] = rinv * (m[0][1] - m[1][0]);
251  dest[3] = r * 0.5f;
252  } else if (m[0][0] >= m[1][1] && m[0][0] >= m[2][2]) {
253  r = sqrtf(1.0f - m[1][1] - m[2][2] + m[0][0]);
254  rinv = 0.5f / r;
255 
256  dest[0] = r * 0.5f;
257  dest[1] = rinv * (m[0][1] + m[1][0]);
258  dest[2] = rinv * (m[0][2] + m[2][0]);
259  dest[3] = rinv * (m[1][2] - m[2][1]);
260  } else if (m[1][1] >= m[2][2]) {
261  r = sqrtf(1.0f - m[0][0] - m[2][2] + m[1][1]);
262  rinv = 0.5f / r;
263 
264  dest[0] = rinv * (m[0][1] + m[1][0]);
265  dest[1] = r * 0.5f;
266  dest[2] = rinv * (m[1][2] + m[2][1]);
267  dest[3] = rinv * (m[2][0] - m[0][2]);
268  } else {
269  r = sqrtf(1.0f - m[0][0] - m[1][1] + m[2][2]);
270  rinv = 0.5f / r;
271 
272  dest[0] = rinv * (m[0][2] + m[2][0]);
273  dest[1] = rinv * (m[1][2] + m[2][1]);
274  dest[2] = r * 0.5f;
275  dest[3] = rinv * (m[0][1] - m[1][0]);
276  }
277 }

306 CGLM_INLINE float glm_mat3_det(mat3 mat) {
307  float a = mat[0][0], b = mat[0][1], c = mat[0][2],
308  d = mat[1][0], e = mat[1][1], f = mat[1][2],
309  g = mat[2][0], h = mat[2][1], i = mat[2][2];
310 
311  return a * (e * i - h * f) - d * (b * i - c * h) + g * (b * f - c * e);
312 }

320 CGLM_INLINE void glm_mat3_inv(mat3 mat, mat3 dest) {
321  float det;
322  float a = mat[0][0], b = mat[0][1], c = mat[0][2],
323  d = mat[1][0], e = mat[1][1], f = mat[1][2],
324  g = mat[2][0], h = mat[2][1], i = mat[2][2];
325 
326  dest[0][0] = e * i - f * h;
327  dest[0][1] = -(b * i - h * c);
328  dest[0][2] = b * f - e * c;
329  dest[1][0] = -(d * i - g * f);
330  dest[1][1] = a * i - c * g;
331  dest[1][2] = -(a * f - d * c);
332  dest[2][0] = d * h - g * e;
333  dest[2][1] = -(a * h - g * b);
334  dest[2][2] = a * e - b * d;
335 
336  det = 1.0f / (a * dest[0][0] + b * dest[1][0] + c * dest[2][0]);
337 
338  glm_mat3_scale(dest, det);
339 }
