173 CGLM_INLINE void glm_vec2_complex_mul(vec2 a, vec2 b, vec2 dest) {
174  float tr, ti;
175  tr = a[0] * b[0] - a[1] * b[1];
176  ti = a[0] * b[1] + a[1] * b[0];
177  dest[0] = tr;
178  dest[1] = ti;
179 }

188 CGLM_INLINE void glm_vec2_complex_div(vec2 a, vec2 b, vec2 dest) {
189  float tr, ti;
190  float const ibnorm2 = 1.0f / (b[0] * b[0] + b[1] * b[1]);
191  tr = ibnorm2 * (a[0] * b[0] + a[1] * b[1]);
192  ti = ibnorm2 * (a[1] * b[0] - a[0] * b[1]);
193  dest[0] = tr;
194  dest[1] = ti;
195 }
